# 创建vue实例

<!DOCTYPE html>
<html lang="en">

<head>
    <title>Vssue Demo</title>
    <link rel="stylesheet" href="https://unpkg.com/vssue/dist/vssue.min.css">
    <script src="https://cdn.jsdelivr.net/npm/vue@2.6.14"></script>
    <script src="https://unpkg.com/vssue/dist/vssue.gitee.min.js"></script>
</head>

<body>
    <div id="vssue"></div>
    <script>
        new Vue({
            el: '#vssue',
            data() {
                return {
                    title: 'Vssue Demo',
                    options: {
                        locale: 'zh',
                        owner: 'guo567',
                        repo: 'vssue-demo',
                        clientId: 'bf7b41119d078a9a0ede19979027047aa6c8a167d4757a79f364798eb0b17cbc',
                        clientSecret: '51775f10632b06480cee9194c2afb583672d97a47246781a0cbf1bbfea4adb55'
                    }
                }
            },
            template: `<vssue :title="title" :options="options"></vssue>`
        })
    </script>
</body>

</html>
